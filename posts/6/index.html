动态区间k-th小数或k-th大数(主席树和树状数组)

ZOJ 2112

主席树和树状数组

不知道主席树的可以去看下我的另一篇文章静态区间k-th小数或k-th大数(主席树)

  • 相同
  1. 建原序列主席树与静态区间主席树一样；
  • 不同
  1. 所修改的值也包括在原序列，一起li’san’hua
  2. 在修改值与查找上增加了树状数组

    BIT是树状数组，计Find(x)表示x在离散化后数组中的下标
    树状数组可以O(logn)计算“前缀和”，在查询静态[l,r]区间的k-th小值时是用[1,r]和[1,l-1]两棵树相减；所以在动态区间中只要用树状数组维护修改的信息，在询问时，除了上面两棵树外，在树状数组上多查询[1,l-1]和[1,r]修改了多少就可以了。
    在修改a[p]值时，将树状数组从p位置开始，每个BIT都在原来的基础上新建一个Find(a[p])对应的分支，该分支的sum-1；接着在修改后的BIT基础上再新建一个Find(z)对应的分支，该分支的sum+1；修改a[p]=z。
    在查询时，将原序列第r和第l-1两棵树和修改的树状数组对应的大约log(l-1)+log(r)棵树作计算就得出[l,r]区间修改后的k-th小数。

#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const int maxn = 1e5+10;
struct Node{
    int l,r,sum;
}T[maxn*40];
int rt[maxn*40],tot;//rt静态主席树,tot所用节点数
int a[maxn],b[maxn],siz;//b是离散化后数组,siz是离散化后个数
int BIT[maxn];        //树状数组
struct command{//存询问和修改的结构体
    char op;
    int L,R,K;
}com[10010];
void build(int &p,int l,int r){//建第0棵树
    p = ++tot;
    T[p].sum = 0;
    if(l == r) return;
    int mid = (l+r)>>1;
    build(T[p].l,l,mid);
    build(T[p].r,mid+1,r);
}
void update(int &p,int pre,int l,int r,int k,int v){
    p = ++tot;//新节点
    T[p] = T[pre];//共用节点，在下一递归中更新l或r其中一个
    T[p].sum += v;//沿途走下去的分支(要更新的点)中都要比上一个版本多或少1
    if(l == r) return;
    int mid = (l+r)>>1;
    if(k <= mid) update(T[p].l,T[pre].l,l,mid,k,v);//在小的一边
    else update(T[p].r,T[pre].r,mid+1,r,k,v);//在大的一边
}
void modify(int x,int k,int v){//维护树状数组，更新修改的相关位置
    for(;x < maxn;x += (x&-x))
        update(BIT[x],BIT[x],1,siz,k,v);//在原来的版本上进行修改 
}
inline int Find(int x){//找离散化后的下标
    return lower_bound(b+1,b+siz+1,x)-b;
}
void C(int p,int z){//修改值
    modify(p,Find(a[p]),-1);//a[p]所在分支-1，去掉a[p]
    modify(p,Find(z),1);//z所在分支+1，加上z
    a[p] = z;//更新值
}
int buf1[20],buf2[20],buf1_cnt,buf2_cnt;
int query(int l,int r,int k){
    if(l == r) return l;
    int t = 0;
    for(int i = 0;i < buf2_cnt;++i)//加上[1,R]区间左边的数量
        t += T[T[buf2[i]].l].sum;
    for(int i = 0;i < buf1_cnt;++i)//减去[1,L-1]区间左边的数量
        t -= T[T[buf1[i]].l].sum;
    for(int i = 0;i < buf1_cnt;++i)//沿分支往线段树下走
        buf1[i] = (k <= t?T[buf1[i]].l:T[buf1[i]].r);
    for(int i = 0;i < buf2_cnt;++i)//沿分支往线段树下走
        buf2[i] = (k <= t?T[buf2[i]].l:T[buf2[i]].r);
    int mid = (l+r)>>1;
    if(k <= t) return query(l,mid,k);//k-th小在小的一边
    else return query(mid+1,r,k-t);//k-th小在大的一边，继续找(k-t)-th小
}
void Q(int L,int R,int K){//查找K-th小
    buf1_cnt = 0;
    buf1[buf1_cnt++] = rt[L-1];//原序列树
    for(int i = L-1;i > 0;i-=(i&-i))
        buf1[buf1_cnt++] = BIT[i];//修改后树
    buf2_cnt=0;
    buf2[buf2_cnt++] = rt[R];//原序列树
    for(int i = R;i > 0;i-=(i&-i))
        buf2[buf2_cnt++] = BIT[i];//修改后树
    int qq = query(1,siz,K);
    printf("%d\n",b[qq]);
}
int main() {
    int t;scanf("%d",&t);
    while(t--){
        int n,q;
        scanf("%d%d",&n,&q);
        for(int i = 1;i <= n;++i){
            cin >> a[i];
            b[i] = a[i];
        } 
        siz = n;
        char ch[2];
        for(int i = 1;i <= q;++i){
            scanf("%s",ch);
            if(ch[0] == 'Q'){
                scanf("%d%d%d",&com[i].L,&com[i].R,&com[i].K);
                com[i].op = 'Q';
            }
            else{
                scanf("%d%d",&com[i].L,&com[i].R);
                com[i].op = 'C';
                b[++siz] = com[i].R;//R是修改后的值
            }
        }
        sort(b+1,b+siz+1);
        siz = unique(b+1,b+siz+1)-b-1;//去重后离散化
        tot = 0;
        build(rt[0],1,siz);//建第0棵树
        for(int i = 1;i <= n;++i){
            int f = Find(a[i]);//找离散化后的下标
            update(rt[i],rt[i-1],1,siz,f,1);//建原序列1-n棵树
        } 
        for(int i = 1;i <= n;++i) //初始化树状数组,最初没有修改信息 
            BIT[i] = rt[0];
        for(int i = 1;i <= q;++i)
            com[i].op == 'Q'?Q(com[i].L,com[i].R,com[i].K):C(com[i].L,com[i].R);
    }
    system("pause");
    return 0;
}
